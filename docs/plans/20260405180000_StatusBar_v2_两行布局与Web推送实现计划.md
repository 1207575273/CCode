/** 采集所有 CPU 核心的 idle/total 时间 */
export function sampleCpuTimes(): { idle: number; total: number } {
  const cpuList = cpus()
  let idle = 0, total = 0
  for (const cpu of cpuList) {
    idle += cpu.times.idle
    total += cpu.times.user + cpu.times.nice + cpu.times.sys + cpu.times.irq + cpu.times.idle
  }
  return { idle, total }
}

      prevSysCpuRef.current = currentSysCpu

      let procCpuPercent = 0
      const now = Date.now()

      if (prevCpuUsageRef.current && prevCpuTimeRef.current > 0) {

          const systemDelta = currentCpu.system - prevCpuUsageRef.current.system
          const totalCpuUs = userDelta + systemDelta
          const elapsedUs = elapsedMs * 1000
          procCpuPercent = (totalCpuUs / elapsedUs / coreCount) * 100

      prevCpuUsageRef.current = currentCpu

重写 `StatusBar` 组件和 `buildSegments`，改为两行渲染：

```tsx
/** SYS 行 段定义 */
function buildSysSegments(data: StatusBarData): Segment[] {
  const segments: Segment[] = []

  // SYS MEM
  const memBar = renderBar(data.sysMemPercent, BAR_WIDTH)
  const memPct = `${Math.round(data.sysMemPercent)}%`.padStart(4)
  const memUsed = formatBytes(data.sysMemUsedBytes)
  const memTotal = formatBytes(data.sysMemTotalBytes)
  const memText = `MEM ${memBar} ${memPct} ${memUsed}/${memTotal}`
  segments.push({
    key: 'sys-mem',
    width: memText.length,
    render: () => (
      <Text>
        <Text dimColor>MEM </Text>
        <Text color={barColor(data.sysMemPercent)}>{memBar}</Text>
        <Text dimColor> {memPct} {memUsed}/{memTotal}</Text>
      </Text>
    ),
  })

  // SYS CPU
  const cpuBar = renderBar(data.sysCpuPercent, BAR_WIDTH)
  const cpuPct = `${Math.round(data.sysCpuPercent)}%`.padStart(4)
  const cpuText = `CPU ${cpuBar} ${cpuPct}`
  segments.push({
    key: 'sys-cpu',
    width: cpuText.length,
    render: () => (
      <Text>
        <Text dimColor>CPU </Text>
        <Text color={barColor(data.sysCpuPercent)}>{cpuBar}</Text>
        <Text dimColor> {cpuPct}</Text>
      </Text>
    ),
  })

  return segments
}

/** PROC 行 段定义 */
function buildProcSegments(
  data: StatusBarData,
  tokenStats: SessionCostStats | null,
  contextState: ContextWindowState | null,
): Segment[] {
  const segments: Segment[] = []

  // PROC MEM
  const memBar = renderBar(data.procMemPercent, BAR_WIDTH)
  const memPct = `${Math.round(data.procMemPercent)}%`.padStart(4)
  const memAbs = formatBytes(data.procMemUsedBytes)
  const memText = `MEM ${memBar} ${memPct} ${memAbs}`
  segments.push({
    key: 'proc-mem',
    width: memText.length,
    render: () => (
      <Text>
        <Text dimColor>MEM </Text>
        <Text color={barColor(data.procMemPercent)}>{memBar}</Text>
        <Text dimColor> {memPct} {memAbs}</Text>
      </Text>
    ),
  })

  // PROC CPU
  const cpuBar = renderBar(data.procCpuPercent, BAR_WIDTH)
  const cpuPct = `${Math.round(data.procCpuPercent)}%`.padStart(4)
  const cpuText = `CPU ${cpuBar} ${cpuPct}`
  segments.push({
    key: 'proc-cpu',
    width: cpuText.length,
    render: () => (
      <Text>
        <Text dimColor>CPU </Text>
        <Text color={barColor(data.procCpuPercent)}>{cpuBar}</Text>
        <Text dimColor> {cpuPct}</Text>
      </Text>
    ),
  })

  // Elapsed
  const elapsed = formatElapsed(data.elapsedMs)
  segments.push({
    key: 'elapsed',
    width: elapsed.length + 3,
    render: () => <Text dimColor>⏱ {elapsed}</Text>,
  })

  // Token
  if (tokenStats && tokenStats.callCount > 0) {
    const tokIn = formatTokenCount(tokenStats.totalInputTokens)
    const tokOut = formatTokenCount(tokenStats.totalOutputTokens)
    const tokText = `${tokIn}/${tokOut} tok`
    segments.push({
      key: 'token',
      width: tokText.length,
      render: () => <Text dimColor>{tokText}</Text>,
    })
  }

  // Context
  if (contextState && contextState.lastInputTokens > 0) {
    const ctxPct = `${(contextState.usedPercentage * 100).toFixed(0)}%`
    const ctxText = `Ctx ${ctxPct}`
    const ctxColor = contextState.level === 'overflow' || contextState.level === 'critical'
      ? 'red' as const
      : contextState.level === 'warning'
        ? 'yellow' as const
        : undefined
    segments.push({
      key: 'context',
      width: ctxText.length,
      render: () => (
        <Text {...(ctxColor ? { color: ctxColor } : { dimColor: true })}>
          {ctxText}
        </Text>
      ),
    })
  }

  // Cost
  if (tokenStats && tokenStats.callCount > 0) {
    const sym = (c: string) => c === 'CNY' ? '¥' : '$'
    const costParts = Object.entries(tokenStats.costByCurrency)
      .filter(([, v]) => v > 0)
      .map(([c, v]) => `${sym(c)}${v.toFixed(4)}`)
    if (costParts.length > 0) {
      const costText = costParts.join('+')
      segments.push({
        key: 'cost',
        width: costText.length,
        render: () => <Text dimColor>{costText}</Text>,
      })
    }
  }

  return segments
}

/** 截断 segments 使其总宽度不超过 maxWidth */
function truncateSegments(segments: Segment[], maxWidth: number): Segment[] {
  const separator = ' | '
  const separatorWidth = separator.length
  const visible: Segment[] = []
  let totalWidth = 0

  for (let i = 0; i < segments.length; i++) {
    const seg = segments[i]
    if (!seg) break
    const segWidth = seg.width + (i > 0 ? separatorWidth : 0)
    if (totalWidth + segWidth > maxWidth) break
    visible.push(seg)
    totalWidth += segWidth
  }
  return visible
}

/** 渲染一行 segments */
function renderLine(segments: Segment[], prefix: string): React.ReactNode {
  const separator = ' | '
  if (segments.length === 0) return null
  return (
    <Box paddingX={1}>
      <Text dimColor>{prefix}</Text>
      {segments.map((seg, i) => (
        <React.Fragment key={seg.key}>
          {i > 0 && <Text dimColor>{separator}</Text>}
          {seg.render()}
        </React.Fragment>
      ))}
    </Box>
  )
}

export function StatusBar({ data, tokenStats, contextState, terminalWidth }: StatusBarProps): React.ReactNode {
  if (!data) return null

  const prefixWidth = 5  // "SYS " 或 "PROC" + 空格
  const maxWidth = terminalWidth - 2 - prefixWidth

  const sysSegments = truncateSegments(buildSysSegments(data), maxWidth)
  const procSegments = truncateSegments(buildProcSegments(data, tokenStats, contextState), maxWidth)

  if (sysSegments.length === 0 && procSegments.length === 0) return null

  return (
    <Box flexDirection="column">
      {renderLine(sysSegments, 'SYS  ')}
      {renderLine(procSegments, 'PROC ')}
    </Box>
  )
}
```

## 实施经验与教训

### 一、需求演进——接受变化，控制代价

本次 StatusBar 经历了三轮需求演进：

| 版本 | 布局 | 触发原因 |
|------|------|----------|
| v1 | 单行（MEM + CPU + 时间 + token + ctx + cost） | 初始方案 |
| v2 | 双行（SYS 资源 + PROC 资源含会话指标） | 用户要求分层展示系统/进程 |
| v3（最终） | 三行（SYS + PROC + INFO） | 用户认为 PROC 行太挤，会话指标独立一行 |

**教训**：StatusBar 这类展示层组件，用户对"看起来怎么样"的判断只能在看到实物后才能确定。前期设计时过早固化布局是浪费的——正确做法是**快速出可运行的原型，让用户看到实物再迭代**。

**好的做法**：
- 数据采集层（`useStatusBar`）和展示层（`StatusBar.tsx`）从一开始就分离，布局变更只改展示层，采集逻辑不动
- Segment 抽象使得"哪些指标放哪一行"只需改 `buildXxxSegments()` 的组合方式，不碰渲染逻辑

### 二、CLI 与 Web 端数据同步——别忘了真正传数据

**踩坑**：v2 实现时 eventBus 推送 `token: null, context: null` 写死了，导致 Web 端 INFO 行永远只显示运行时间。问题直到用户在 Web 端实际看到才发现。

**根因**：CLI 端 StatusBar 的 token/context 数据是通过 `App.tsx` props 注入的（事件驱动），而 eventBus 推送逻辑在 `useStatusBar` hook 中——两条路径不一致。设计文档写了"token/context 由调用方注入"但没明确**推送时也要注入**。

**修复**：给 `useStatusBar` 加 `getTokenStats` / `getContextState` 回调，推送时调用。

**教训**：
- 当数据有两个消费端（CLI 渲染 + Web 推送）时，必须确保两条路径获得相同的数据
- 推送层不能假设"下游会补充"——**谁推送谁负责数据完整性**
- 设计文档中写 `token: null` 时就该红灯亮——这不是占位符，这是 bug

### 三、闭包陈旧——React Hook 的经典陷阱

**踩坑**：`useStatusBar` 中 `setInterval` 的回调捕获了 `getTokenStats` / `getContextState` / `accumulatedMs` / `sessionStartTime`，但这些值在 props 变化后不会更新（闭包捕获的是旧值）。

**表现**：
- resume 后 `accumulatedMs` 变了但时间不累加
- LLM 调用后 `getTokenStats()` 返回的是旧快照

**修复**：所有在 `setInterval` 回调中访问的变量改用 `useRef` 持有最新值，回调通过 `.current` 读取。

```typescript
// ✗ 闭包陈旧
useEffect(() => {
  const id = setInterval(() => {
    doSomething(accumulatedMs)  // 永远是初始值
  }, 3000)
  return () => clearInterval(id)
}, [])  // 空依赖 = 永远不重建

// ✓ ref 持有最新值
const accMsRef = useRef(accumulatedMs)
accMsRef.current = accumulatedMs  // 每次渲染更新
useEffect(() => {
  const id = setInterval(() => {
    doSomething(accMsRef.current)  // 始终读到最新值
  }, 3000)
  return () => clearInterval(id)
}, [])
```

**教训**：任何在 `setInterval` / `setTimeout` / 事件监听器中访问的外部状态，如果可能变化，都必须用 ref。这不是优化，是正确性问题。

### 四、防御式编程——面向失败设计

打磨阶段发现的防御缺失（均已修复）：

| 问题 | 失败场景 | 修复 |
|------|----------|------|
| `sample()` 无 try-catch | 极端内存压力下 `process.memoryUsage()` 可能抛异常 | 整体 try-catch，跳过本次采样 |
| `sysUsedMem` 可能为负 | `freemem()` 实时取但 `totalmem()` 是缓存值，虚拟化环境可能不一致 | `Math.max(0, totalMem - free)` |
| `cpuCountRef` 可能为 0 | 极端环境 `os.cpus()` 返回空数组 | `cpus().length \|\| 1` 兜底至少 1 核，防止除零 |
| `clamp100` 未防御 NaN | 除零或 API 返回异常值产生 NaN | `if (!Number.isFinite(v)) return 0` |
| `formatBytes` / `formatElapsed` 负数入参 | 上游计算错误传入负值 | 负数兜底为 `0MB` / `00:00` |
| `renderBar` 未自行 clamp | 展示组件依赖上游一定传 0-100 | 组件内自己 `Math.min(100, Math.max(0, ...))` |
| `getLastAccumulatedMs` 全量读文件 | 长会话 JSONL 几十 MB，全量 `readFileSync` 占内存 | 只读尾部 4KB |

**原则**：纯展示组件不能假设输入一定合法——**每一层都做自己的防御**。上游 clamp 了不代表展示层不需要 clamp，因为中间可能还有其他调用路径。

### 五、"启动即展示"的正确做法

**初始设计**：`hasMessages` 为 false 时不渲染、不采集。用户反馈"应该一开始就展示"。

**问题**：不仅是展示时机的问题——`hasMessages` 作为 guard 导致 hook 内部 `useEffect` 被跳过，首次采样延迟到有消息时才执行。改为 `enabled`（由 config 开关控制）后，启动即采样、即渲染。

**教训**：Hook 的 early return guard 要谨慎——它跳过的不只是渲染，还有所有 useEffect 的注册。如果 guard 条件可能频繁切换（如 `hasMessages`），会导致定时器反复创建销毁。用一个稳定的 `enabled` flag 更可靠。

### 六、Web 端的二等公民问题

实现过程中 Web 端始终是"后补"的：先做 CLI，再想 Web。这导致：

1. eventBus 推送写死 `null`（CLI 不走这条路径所以没发现）
2. Web StatusBar 组件的三行布局需要单独再改一遍
3. ChatPage 原有的 context badge 忘了同步移除

**教训**：如果一个功能承诺 CLI + Web 双端，**设计阶段就应该画两端的数据流图**，确保推送协议完整。实现时 CLI 和 Web 的适配应该在同一个 Task 中完成，而不是 CLI 全做完再"补一个 Web 的"。

### 七、工程质量检查清单

基于本次经验，形成 StatusBar 类实时展示功能的 review checklist：

- [ ] 定时器回调中是否有闭包陈旧风险？（检查所有 `setInterval` 内访问的外部变量）
- [ ] 数据采集 API（`process.*` / `os.*`）是否有 try-catch？
- [ ] 数值计算是否防御了 NaN / Infinity / 负数 / 除零？
- [ ] 展示组件是否自行做了入参防御（不依赖上游 clamp）？
- [ ] 多端推送的数据是否完整（不是 null 占位符）？
- [ ] 文件 I/O 是否考虑了大文件场景（避免全量读入内存）？
- [ ] `useEffect` 的 cleanup 是否清理了所有定时器和事件监听器？
- [ ] 配置开关为 false 时是否真正零开销（不启动定时器、不推送）？
